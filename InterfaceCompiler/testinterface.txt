interface LemonWM {
    CreateWindow(string title, s32 x, s32 y, s32 width, s32 height, u32 flags) -> (s64 status, s64 bufferKey)
    DestroyWindow()

    SetTitle(string title)
    Relocate(s32 x, s32 y)
    Resize(s32 width, s32 height) -> (s64 status, s64 bufferKey)
    Minimize(s64 windowID, bool minimized)
    DisplayContextMenu(s32 x, s32 y, string serializedEntries)
    InitializeShellConnection()
}
